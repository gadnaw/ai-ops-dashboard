# Plan 03-01: Prompt Manager Service

## Objective

Build the complete server-side foundation for prompt version management. This plan creates the database schema, PostgreSQL trigger for per-template auto-incrementing version numbers, FK constraint migration, server actions for creating and rolling back versions, REST API endpoints, and extends the existing `/api/v1/chat` route to resolve prompt versions at request time.

No UI is built here. The output is the data layer and API surface that Plans 03-02 and 03-03 build on top of.

**Purpose:** The prompt engineering workflow loop requires a reliable, immutable versioning system on the server before any UI can be layered over it. Version numbers must be unique per template (not globally), content must be write-once, and rollback must be a single atomic DB update — not an application-layer copy.

**Output:** Working server-side prompt versioning with REST endpoints testable via curl. The `/api/v1/chat` route becomes version-aware: pass `promptVersionId` in the request body and the route resolves the version's content as the system prompt, logging `prompt_version_id` on every request.

```prisma
  @@map("prompt_templates")
}

model PromptVersion {
  id           String   @id @default(cuid())
  templateId   String   @map("template_id")

  content      String   @db.Text
  systemPrompt String?  @map("system_prompt") @db.Text
  modelConfig  Json     @default("{}") @map("model_config")

  template     PromptTemplate  @relation("TemplateVersions", fields: [templateId], references: [id], onDelete: Cascade)
  // A version can be the active version for at most one template
  activeFor    PromptTemplate? @relation("ActiveVersion")

  @@unique([templateId, version])
  @@index([templateId, createdAt(sort: Desc)])
```

```sql
-- =============================================================================
-- Phase 3: Prompt Versioning — Custom SQL
-- =============================================================================

-- 1. Immutability trigger: prevent UPDATE on version content columns
CREATE OR REPLACE FUNCTION enforce_prompt_version_immutability()
RETURNS TRIGGER AS $$
BEGIN
  -- Allow updates to non-content columns (none currently, but future-proof)
  IF OLD.content IS DISTINCT FROM NEW.content
     OR OLD.system_prompt IS DISTINCT FROM NEW.system_prompt
     OR OLD.model_config IS DISTINCT FROM NEW.model_config
  THEN
    RAISE EXCEPTION 'prompt_versions content is immutable. Create a new version instead.';
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER prompt_version_immutability
  BEFORE UPDATE ON prompt_versions
  FOR EACH ROW
  EXECUTE FUNCTION enforce_prompt_version_immutability();

-- 2. Per-template version auto-increment trigger with advisory lock
--    Prisma INSERTs version=0; this trigger overwrites with MAX(version)+1 for that template.
--    pg_advisory_xact_lock serializes concurrent INSERTs for the same template_id.
--    IMPORTANT: template_id is CUID (text), not bigint. We hash it to bigint for advisory lock.
CREATE OR REPLACE FUNCTION assign_prompt_version()
RETURNS TRIGGER AS $$
DECLARE
  next_version INTEGER;
  lock_key     BIGINT;
BEGIN
  -- Hash the CUID template_id to a stable bigint for the advisory lock
  -- hashtext() is a stable PostgreSQL built-in (no extension required)
  lock_key := hashtext(NEW.template_id::text)::bigint;
  PERFORM pg_advisory_xact_lock(lock_key);

  SELECT COALESCE(MAX(version), 0) + 1
  INTO next_version
  FROM prompt_versions
  WHERE template_id = NEW.template_id;

  NEW.version := next_version;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_prompt_version
  BEFORE INSERT ON prompt_versions
  FOR EACH ROW
  EXECUTE FUNCTION assign_prompt_version();

-- 3. Add FK constraint: request_logs.prompt_version_id -> prompt_versions(id)
--    Phase 2 added this column WITHOUT FK (prompt_versions table didn't exist yet).
--    Phase 3 migration adds the FK constraint now that the table exists.
--    ON DELETE SET NULL: deleting a prompt version nullifies the FK in request_logs
--    (historical log data is preserved even if template is deleted).
ALTER TABLE request_logs
  ADD CONSTRAINT fk_prompt_version
  FOREIGN KEY (prompt_version_id)
  REFERENCES prompt_versions(id)
  ON DELETE SET NULL;

-- 4. RLS on prompt tables (Prisma bypasses with service role — defense-in-depth only)
ALTER TABLE prompt_templates ENABLE ROW LEVEL SECURITY;
ALTER TABLE prompt_versions ENABLE ROW LEVEL SECURITY;

-- Service role bypasses RLS. Client-side reads (if any future use) are read-only.
CREATE POLICY "prompt_templates_select_all"
  ON prompt_templates FOR SELECT USING (true);

CREATE POLICY "prompt_versions_select_all"
  ON prompt_versions FOR SELECT USING (true);

-- 5. Seed prompt templates for demo (matches Phase 2 seed data prompt_version_ids)
-- Insert 3 named templates matching the slug patterns used in seedBaseData()
```

```typescript
// lib/prompts/variables.ts
// No "use server" or "use client" directives — safe to import from both contexts.

/**
 * Extract all {{variableName}} references from a prompt template content string.
 * Returns a sorted, deduplicated array of variable names.
 *
 * Rules:
 * - Matches: {{name}}, {{order_id}}, {{_private}}, {{camelCase}}
 * - Does NOT match: {{{triple}}}, {{#block}}, {{/block}}, {{123starts_with_digit}}
 * - Spaces inside braces are NOT matched (intentional strictness)
 */
export function extractVariables(content: string): string[] {
  const regex = /\{\{([a-zA-Z_][a-zA-Z0-9_]*)\}\}/g;
  const variables = new Set<string>();
  let match: RegExpExecArray | null;

  while ((match = regex.exec(content)) !== null) {
    variables.add(match[1]!);
  }

  return Array.from(variables).sort();
}

/**
 * Replace {{variableName}} placeholders with provided values.
 * Variables not in the values map are left as-is: {{unreplaced}}
 */
export function interpolateVariables(
  content: string,
  values: Record<string, string>
): string {
  return content.replace(
    /\{\{([a-zA-Z_][a-zA-Z0-9_]*)\}\}/g,
    (_, key: string) => values[key] ?? `{{${key}}}`
  );
}
```

```typescript
/**
 * List all prompt templates with their active version (if set).

export async function getTemplates() {
  return prisma.promptTemplate.findMany({
    orderBy: { createdAt: 'desc' },
    include: {
      activeVersion: {
        select: {
          id: true,
          version: true,
          content: true,
          systemPrompt: true,
          variables: true,
          createdAt: true,
        },
      },
      versions: {
        select: { id: true, version: true, createdAt: true },
        orderBy: { version: 'desc' },
        take: 1, // Just the latest for the list view
      },
    },
  });
}

/**
 * Get a single template by slug with all versions.
 * Returns null if not found.
 */
export async function getTemplateWithVersions(slug: string) {
  return prisma.promptTemplate.findUnique({
    where: { slug },
    include: {
      activeVersion: true,
      versions: {
        orderBy: { version: 'desc' },
      },
    },
  });
}

/**
 * Get a single prompt version by ID.
 * Returns null if not found.
 */
export async function getVersion(versionId: string) {
  return prisma.promptVersion.findUnique({
    where: { id: versionId },
    include: {
      template: { select: { id: true, slug: true, name: true } },
    },
  });
}

/**
 * Get two specific versions for diff view.
 * Returns both or throws if either is not found.
 */
export async function getTwoVersionsForDiff(
  versionId1: string,
  versionId2: string
) {
  const [v1, v2] = await Promise.all([
    prisma.promptVersion.findUnique({ where: { id: versionId1 } }),
    prisma.promptVersion.findUnique({ where: { id: versionId2 } }),
  ]);

  if (!v1 || !v2) {
    throw new Error('One or both versions not found');
  }
  return { v1, v2 };
}
```

**Step 2 — Create src/lib/prompts/actions.ts:**

```typescript
// lib/prompts/actions.ts
'use server';

import { prisma } from '@/lib/db/prisma';
import { revalidatePath } from 'next/cache';
import { extractVariables } from './variables';

/**
 * Create a new prompt template.
 * Slug must be unique — throws Prisma unique constraint error if duplicate.
 */
export async function createPromptTemplate(data: {
  slug: string;
  name: string;
  description?: string;
}) {
  const template = await prisma.promptTemplate.create({
    data: {
      slug: data.slug,
      name: data.name,
      description: data.description ?? null,
    },
  });

  revalidatePath('/prompts');
  return template;
}

/**
 * Create a new immutable version of an existing template.
 *
 * IMPORTANT: version number is set by the PostgreSQL trigger (assign_prompt_version).
 * We INSERT with version=0; the trigger overwrites with MAX(version)+1 per template.
 * Do NOT attempt to compute version numbers in application code.
 *
 * After creation, if no active version exists, automatically sets this as active.
 */
export async function createPromptVersion(
  templateId: string,
  data: {
    content: string;
    systemPrompt?: string;
    modelConfig?: Record<string, unknown>;
  }
) {
  const variables = extractVariables(data.content);

  const version = await prisma.promptVersion.create({
    data: {
      templateId,
      version: 0, // PostgreSQL trigger (assign_prompt_version) overwrites this on INSERT
      content: data.content,
      systemPrompt: data.systemPrompt ?? null,
      modelConfig: data.modelConfig ?? {},
      variables,
    },
  });

  // If template has no active version, set this as the active version
  const template = await prisma.promptTemplate.findUnique({
    where: { id: templateId },
    select: { activeVersionId: true },
  });

  if (!template?.activeVersionId) {
    await prisma.promptTemplate.update({
      where: { id: templateId },
      data: { activeVersionId: version.id },
    });
  }

  revalidatePath('/prompts');
  return version;
}

/**
 * Roll back a template to a specific prior version.
 * Validates that versionId belongs to templateId before updating.
 * Uses optimistic UI pattern — client updates immediately, this confirms.
 */
export async function rollbackToVersion(
  templateId: string,
  versionId: string
) {
  // Verify the version belongs to this template (prevents cross-template rollback)
  const version = await prisma.promptVersion.findFirst({
    where: { id: versionId, templateId },
    select: { id: true, version: true },
  });

  if (!version) {
    throw new Error('Version not found or does not belong to this template');
  }

  await prisma.promptTemplate.update({
    where: { id: templateId },
    data: { activeVersionId: versionId },
  });

  revalidatePath('/prompts');
  revalidatePath(`/prompts/[slug]`, 'page');
  return { success: true, activeVersionId: versionId };
}

/**
 * Delete a prompt template and all its versions.
 * Cascading delete — prompt_versions.on_delete = Cascade.
 * request_logs.prompt_version_id becomes NULL for orphaned logs (ON DELETE SET NULL FK).
 */
export async function deletePromptTemplate(templateId: string) {
  await prisma.promptTemplate.delete({
    where: { id: templateId },
  });

  revalidatePath('/prompts');
}
```

```typescript
import { NextResponse } from 'next/server';
import { rollbackToVersion } from '@/lib/prompts/actions';

/**
 * POST /api/v1/prompts/[id]/rollback
 * Roll back a template's active version.
 * Body: { versionId: string }
 *
 * Used by the VersionList Client Island via RollbackButton (in addition to Server Action direct calls).
 */
export async function POST(
  request: Request,
  { params }: { params: { id: string } }
) {
  try {
    const body = await request.json() as { versionId?: string };

    if (!body.versionId) {
      return NextResponse.json(
        { error: 'versionId is required' },
        { status: 400 }
      );
    }

    const result = await rollbackToVersion(params.id, body.versionId);
    return NextResponse.json(result);
  } catch (error) {
    const message = error instanceof Error ? error.message : 'Rollback failed';
    return NextResponse.json({ error: message }, { status: 400 });
  }
}
```

1. Look up the version's content and systemPrompt from the database
2. Use the version's content as the system prompt (if no explicit systemPrompt in modelConfig)
3. Pass `promptVersionId` to the logger so it appears in request_logs

```typescript
// app/api/v1/chat/route.ts
import { streamText } from 'ai';
import { after } from 'next/server';
import { registry } from '@/lib/model-router/registry';
import { prisma } from '@/lib/db/prisma';

export const maxDuration = 30;

export async function POST(request: Request) {
  const body = await request.json() as {
    prompt?: string;
    messages?: Array<{ role: string; content: string }>;
    promptVersionId?: string;
    modelId?: string;
    modelConfig?: {
      temperature?: number;
      maxTokens?: number;
      systemPrompt?: string;
    };
    // Legacy field — endpoint-based routing (Phase 2 model router)
    endpoint?: string;
  };

  const {
    prompt,
    messages,
    promptVersionId,
    modelId = 'openai:gpt-4o-mini',
    modelConfig = {},
    endpoint,
  } = body;

  // If promptVersionId provided, resolve system prompt from version content
  let resolvedSystemPrompt = modelConfig.systemPrompt;

  if (promptVersionId) {
    try {
      const version = await prisma.promptVersion.findUnique({
        where: { id: promptVersionId },
        select: { content: true, systemPrompt: true },
      });
      if (version) {
        // Version's systemPrompt takes precedence if set, else use content as system prompt
        resolvedSystemPrompt = version.systemPrompt ?? version.content;
      }
    } catch {
      // Non-fatal — proceed without version content
      console.error('Failed to resolve promptVersionId:', promptVersionId);
    }
  }

  // If endpoint provided (legacy Phase 2 path), delegate to the existing streamWithFallback router
  // If modelId provided (Phase 3 playground path), use registry directly
  const result = streamText({
    model: registry.languageModel(modelId),
    prompt: prompt ?? undefined,
    messages: (messages as Parameters<typeof streamText>[0]['messages']) ?? undefined,
    system: resolvedSystemPrompt,
    temperature: modelConfig.temperature ?? 0.7,
    maxOutputTokens: modelConfig.maxTokens ?? 2048,

    onFinish: async ({ text, usage, finishReason }) => {
      after(async () => {
        try {
          await prisma.requestLog.create({
            data: {
              provider: modelId.split(':')[0] ?? 'openai',
              model: modelId.split(':')[1] ?? 'gpt-4o-mini',
              promptVersionId: promptVersionId ?? null,
              inputTokens: usage?.inputTokens ?? 0,
              outputTokens: usage?.outputTokens ?? 0,
              durationMs: 0, // Populated by the existing after() logger pattern
              costUsd: 0,    // Populated by CostCalculator in existing logger
              status: 'success',
              finishReason: finishReason ?? 'unknown',
              responseText: text,
              requestedAt: new Date(),
            },
          });
        } catch (err) {
          console.error('Request log write failed (non-fatal):', err);
        }
      });
    },
  });

  // Constraint H4: Use toUIMessageStreamResponse() — compatible with useCompletion.
  // If useCompletion sends streamProtocol: 'text' and this causes format mismatch,
  // fall back to: return result.toTextStreamResponse();
  return result.toUIMessageStreamResponse();
}
```

```typescript
import { describe, it, expect } from 'vitest';
import { extractVariables, interpolateVariables } from '../variables';

describe('extractVariables', () => {
  it('extracts single variable', () => {
    expect(extractVariables('Hello {{name}}!')).toEqual(['name']);
  });

  it('extracts multiple variables in sorted order', () => {
    expect(extractVariables('{{order_id}} for {{customer_name}}')).toEqual([
      'customer_name',
      'order_id',
    ]);
  });

  it('deduplicates repeated variables', () => {
    expect(extractVariables('{{a}} and {{a}} again')).toEqual(['a']);
  });

  it('returns empty array for no variables', () => {
    expect(extractVariables('No variables here')).toEqual([]);
  });

  it('does NOT match triple braces', () => {
    expect(extractVariables('{{{triple}}}')).toEqual([]);
  });

  it('does NOT match variables starting with digit', () => {
    expect(extractVariables('{{123bad}}')).toEqual([]);
  });

  it('does NOT match Handlebars block helpers', () => {
    expect(extractVariables('{{#if condition}}{{/if}}')).toEqual([]);
  });

  it('matches variables with underscores', () => {
    expect(extractVariables('{{first_name}} {{last_name}}')).toEqual([
      'first_name',
      'last_name',
    ]);
  });
});

describe('interpolateVariables', () => {
  it('replaces known variables', () => {
    expect(
      interpolateVariables('Hello {{name}}!', { name: 'Alice' })
    ).toBe('Hello Alice!');
  });

  it('leaves unknown variables as-is', () => {
    expect(
      interpolateVariables('Hello {{name}} {{unknown}}!', { name: 'Alice' })
    ).toBe('Hello Alice {{unknown}}!');
  });

  it('handles empty values object', () => {
    expect(interpolateVariables('Hello {{name}}!', {})).toBe(
      'Hello {{name}}!'
    );
  });
});
```
